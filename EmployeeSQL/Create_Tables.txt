CREATE TABLE titles(
	title_id VARCHAR NOT NULL,
	title VARCHAR NOT NULL,
	PRIMARY KEY (title_id)
);
-- 7 unique emp_title_id

CREATE TABLE departments(
	dept_no VARCHAR NOT NULL,
	dept_name VARCHAR NOT NULL,
	PRIMARY KEY (dept_no)
);
-- 9 unique dept_no

CREATE TABLE employees(
	emp_no INTEGER NOT NULL,
	emp_title_id VARCHAR NOT NULL,
	birth_date VARCHAR NOT NULL,
	first_name VARCHAR NOT NULL,
	last_name VARCHAR NOT NULL,
	sex VARCHAR NOT NULL,
	hire_date VARCHAR NOT NULL,
	PRIMARY KEY (emp_no)									-- SHOULD THERE BE A ","?
	FOREIGN KEY (emp_title_id) REFERENCES titles(title_id); -- SHOULD THERE BE A ";"?
);
-- 30024 unique emp_no
-- 7 unique emp_title_id, all emp_no have an emp_title_id

CREATE TABLE dept_emp (
	emp_no INTEGER NOT NULL,
	dept_no VARCHAR NOT NULL,
	PRIMARY KEY (emp_no, dept_no), 							-- SHOULD THERE BE A ","?
	FOREIGN KEY (emp_no) REFERENCES employees(emp_no);		-- SHOULD THERE BE A ";"?
	FOREIGN KEY (dept_no) REFERENCES departments(dept_no);	-- SHOULD THERE BE A ";"?

);

-- 30024 unique emp_no, all rows are assigned an emp_no
-- 9 unique dept_no, all emp_no have a dept_no
